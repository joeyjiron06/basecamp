This is my Tailwind config that includes DaisyUI and adds a bit more extra config that I use.

How do you install tailwind and DaisyUI?

```bash
pnpm i --save tailwindcss @tailwindcss/typography autoprefixer daisyui
```

Setup the config file

```javascript
/** @type {import('tailwindcss').Config} */
export default {
  content: ['./src/**/*.{astro,html,js,jsx,md,mdx,svelte,ts,tsx,vue}'],
  theme: {
    extend: {
      maxWidth: {
        // useful utility for setting max page width on components
        'page': '72rem',
      },

      typography: (theme) => ({
        DEFAULT: {
          css: {
            color: theme('colors.base-content'),
          },
        },
      }),
    }
  },

  daisyui: {
    themes: [
      {
        dark: {
          ...require("daisyui/src/theming/themes")["black"],

          primary: "white",
          secondary: "grey",

          "base-100": "#000000",
          "base-200": "#0A0A0A",
          "base-300": "#1A1A1A",
          "base-content": "#E6E6E6",

          // border radius rounded-box utility class, used in card and other large boxes
          "--rounded-box": "0.5rem", 
          // border radius rounded-btn utility class, used in buttons and similar element
          "--rounded-btn": "0.25rem", 
          // border radius rounded-badge utility class, used in badges and similar
          "--rounded-badge": "1.9rem",
          // duration of animation when you click on button 
          "--animation-btn": "0.25s", 
          // duration of animation for inputs like checkbox, toggle, radio, etc
          "--animation-input": "0.2s", 
          // scale transform of button when you focus on it
          "--btn-focus-scale": "0.95", 
          // border width of buttons
          "--border-btn": "1px", 
          // border width of tabs
          "--tab-border": "1px", 
          // border radius of tabs
          "--tab-radius": "0.25rem", 
        },
      },
    ],
  },
  plugins: [
    require('@tailwindcss/typography'),
    require("daisyui")
  ],
}
```
